   Nullable, no CHECK — the default behavior is "no canonical override" and the metadata helper

    -- Pattern: must start with a lowercase letter, then lowercase alnum + single hyphens.
    -- Length 3..120 (sane URL bounds).
    --
    -- CHECK enforces slug shape (must start with a letter, only lowercase alnum + hyphens).
    -- The audit-#38 ms-suffix slug pattern (`error-1778151609106`) is NOT closed by this
    -- regex alone — the n8n flow MUST NOT generate Date.now()-based fallback slugs. The
    -- DB validation triggers gate-9 (DocuSeal mention count) and gate-3 (persona phrase)
    -- further reduce the chance of malformed n8n output reaching the DB. The regex's
    -- specific mitigation here is rejecting numeric-only slugs (e.g., `1234567890`) and
    -- slugs with uppercase / spaces / underscores.

    ALTER TABLE public.blogs DROP CONSTRAINT IF EXISTS blogs_slug_format_check;
    ALTER TABLE public.blogs
      ADD CONSTRAINT blogs_slug_format_check
      CHECK (slug ~ '^[a-z][a-z0-9]*(-[a-z0-9]+)*$' AND length(slug) BETWEEN 3 AND 120);

    **File 4: `phase_6_validation_triggers.sql`**

    ```sql
    -- Phase 6 / BLOG-01: 9 BEFORE-INSERT validation gates as ONE trigger function.
    -- Defense-in-depth — n8n flow (Plan 06-03) runs the same gates pre-flight; DB is the
    -- source of truth and the last guard.

    CREATE OR REPLACE FUNCTION public.validate_blog_post()
    RETURNS trigger
    LANGUAGE plpgsql
    SECURITY DEFINER
    SET search_path = public
    AS $$
    DECLARE
      v_word_count integer;
      v_h2_count integer;
      v_docuseal_count integer;
      v_banned text;
      v_banlist text[] := ARRAY[
        'rent collection', 'online rent', 'autopay', 'auto-pay', 'tenant portal',
        'automated rent', 'collect rent', 'rent processing', 'pay rent online',
        'online payments', 'online rent payment', 'rent collection software',
        'tenants can pay', 'pay rent through',
        'automated workflow', 'rent tracking', 'mobile app access',
        'record rent', 'paid rent', 'pay rent'
      ];
      v_categories text[] := ARRAY['lease-law', 'tax-prep', 'tenant-screening', 'maintenance', 'software-vault'];
    BEGIN
      -- Only run gates on INSERT, and only when status is being set to a published-trail value.
      -- 'archived' rows can be loose; 'draft' rows are operator-controlled. Apply gates to
      -- 'in-review' and 'published' transitions because both face the public.
      IF (TG_OP = 'INSERT' AND NEW.status IN ('in-review', 'published'))
         OR (TG_OP = 'UPDATE' AND NEW.status IN ('in-review', 'published') AND OLD.status NOT IN ('in-review', 'published'))
      THEN
        -- Gate 1: word count 1,200–3,000
        v_word_count := array_length(regexp_split_to_array(coalesce(NEW.content, ''), '\s+'), 1);
        IF v_word_count IS NULL OR v_word_count &lt; 1200 OR v_word_count &gt; 3000 THEN
          RAISE EXCEPTION 'word_count out of range: % (must be 1200..3000)', v_word_count
            USING ERRCODE = '23514';
        END IF;

        -- Gate 2: H2 count 4–10 (markdown ^## )
        -- Single, correct form using regexp_matches() row-counting. (Earlier draft had a
        -- dead `array_length(regexp_matches(..., 'gn'), 1)` line that was immediately
        -- overwritten — Blocker-#3 cleanup removes it.)
        v_h2_count := coalesce(
          (SELECT count(*)::int FROM regexp_matches(coalesce(NEW.content, ''), '^## ', 'gn')),
          0
        );
        IF v_h2_count &lt; 4 OR v_h2_count &gt; 10 THEN
          RAISE EXCEPTION 'h2_count out of range: % (must be 4..10)', v_h2_count
            USING ERRCODE = '23514';
        END IF;

        -- Gate 3: persona phrase — body must contain "landlord" (case-insensitive)
        IF position('landlord' in lower(coalesce(NEW.content, ''))) = 0 THEN
          RAISE EXCEPTION 'persona phrase missing: content must contain "landlord"'
            USING ERRCODE = '23514';
        END IF;

        -- Gate 4: slug pattern — duplicates the CHECK constraint defense-in-depth.
        -- (CHECK runs before trigger; this is here for clearer error messaging on UPDATE.)
        IF NEW.slug !~ '^[a-z][a-z0-9]*(-[a-z0-9]+)*$' OR length(NEW.slug) &lt; 3 OR length(NEW.slug) &gt; 120 THEN
          RAISE EXCEPTION 'slug pattern invalid: % (must match ^[a-z][a-z0-9]*(-[a-z0-9]+)*$, length 3..120)', NEW.slug
            USING ERRCODE = '23514';
        END IF;

        -- Gate 5: meta_description 50–160 chars
        IF NEW.meta_description IS NULL OR length(NEW.meta_description) &lt; 50 OR length(NEW.meta_description) &gt; 160 THEN
          RAISE EXCEPTION 'meta_description length out of range: % (must be 50..160)',
            coalesce(length(NEW.meta_description)::text, 'NULL')
            USING ERRCODE = '23514';
        END IF;

        -- Gate 6: excerpt 80–200 chars
        IF NEW.excerpt IS NULL OR length(NEW.excerpt) &lt; 80 OR length(NEW.excerpt) &gt; 200 THEN
          RAISE EXCEPTION 'excerpt length out of range: % (must be 80..200)',
            coalesce(length(NEW.excerpt)::text, 'NULL')
            USING ERRCODE = '23514';
        END IF;

        -- Gate 7: category enum
        IF NEW.category IS NULL OR NOT (NEW.category = ANY(v_categories)) THEN
          RAISE EXCEPTION 'category not in enum: % (must be one of %)',
            coalesce(NEW.category, 'NULL'), array_to_string(v_categories, ', ')
            USING ERRCODE = '23514';
        END IF;

        -- Gate 8: banlist — content must not contain any banned phrase (case-insensitive)
        FOREACH v_banned IN ARRAY v_banlist LOOP
          IF position(lower(v_banned) in lower(coalesce(NEW.content, ''))) &gt; 0 THEN
            RAISE EXCEPTION 'banlist hit: % (Phase 4 banlist phrase found in content)', v_banned
              USING ERRCODE = '23514';
          END IF;
        END LOOP;

        -- Gate 9: DocuSeal mention count ≤ 1 (Phase 4 COPY-04 de-amp)
        v_docuseal_count := coalesce(
          (SELECT count(*)::int FROM regexp_matches(coalesce(NEW.content, ''), 'DocuSeal', 'gi')),
          0
        );
        IF v_docuseal_count &gt; 1 THEN
          RAISE EXCEPTION 'DocuSeal mention count too high: % (max 1 per Phase 4 COPY-04)', v_docuseal_count
            USING ERRCODE = '23514';
        END IF;
      END IF;

      RETURN NEW;
    END;
    $$;

    REVOKE EXECUTE ON FUNCTION public.validate_blog_post() FROM anon, authenticated;

    DROP TRIGGER IF EXISTS validate_blog_post_trigger ON public.blogs;
    CREATE TRIGGER validate_blog_post_trigger
      BEFORE INSERT OR UPDATE ON public.blogs
      FOR EACH ROW
      EXECUTE FUNCTION public.validate_blog_post();
    ```

    import { createClient, type SupabaseClient } from '@supabase/supabase-js'
    import { afterAll, beforeAll, describe, expect, it } from 'vitest'

    const SLUG_SUFFIX = Date.now().toString()
    const SLUG_PREFIX = `phase-6-rls-test-${SLUG_SUFFIX}`

    function validBody(overrides: Record&lt;string, unknown&gt; = {}) {
      const body =
        '# H1\n\n' +
        'Intro para about landlords with 1–15 rentals. '.repeat(20) +

        title: 'Phase 6 RLS Test Post',
        slug: `${SLUG_PREFIX}-valid`,
        excerpt: 'A test excerpt about landlords with 1–15 rentals to satisfy the excerpt length gate cleanly.',
        content: body,
        category: 'lease-law',
        meta_description: 'Test meta description about landlord lease law content with sufficient length to pass the meta gate.',
        status: 'in-review',
        ...overrides,
      }
    }

    describe('blogs — status workflow + 9 validation gates', () => {
      let service: SupabaseClient
      let anon: SupabaseClient
      const insertedSlugs: string[] = []

      beforeAll(async () => {
        const url = process.env.NEXT_PUBLIC_SUPABASE_URL!
        service = createClient(url, process.env.SUPABASE_SERVICE_ROLE_KEY!)
        anon = createClient(url, process.env.NEXT_PUBLIC_SUPABASE_PUBLISHABLE_KEY!)
        // Cleanup-on-rerun safeguard (Phase 65 SLUG_SUFFIX pattern)
        await service.from('blogs').delete().like('slug', 'phase-6-rls-test-%')
      })

      afterAll(async () => {
        for (const slug of insertedSlugs) {
          await service.from('blogs').delete().eq('slug', slug)
        }
      })

      it('accepts valid in-review payload', async () => {
        const body = validBody({ slug: `${SLUG_PREFIX}-valid` })
        insertedSlugs.push(body.slug as string)
        const { data, error } = await service.from('blogs').insert(body).select('id, status').single()
        expect(error).toBeNull()
        expect(data?.status).toBe('in-review')
      })

      it("rejects status='not-a-state'", async () => {
        const body = validBody({ slug: `${SLUG_PREFIX}-bad-status`, status: 'not-a-state' })
        const result = await service.from('blogs').insert(body)
        expect(result.error).toMatchObject({ message: expect.stringContaining('blogs_status_check') })
      })

      it("rejects slug='1234567890' (numeric-only fails leading-letter regex)", async () => {
        const body = validBody({ slug: '1234567890' })
        const result = await service.from('blogs').insert(body)
        expect(result.error).toMatchObject({ message: expect.stringContaining('blogs_slug_format_check') })
      })

      it("accepts slug='valid-slug-format' (valid kebab-case)", async () => {
        const slug = `${SLUG_PREFIX}-valid-slug-format`
        const body = validBody({ slug })
        insertedSlugs.push(slug)
        const { data, error } = await service.from('blogs').insert(body).select('id').single()
        expect(error).toBeNull()
        expect(data?.id).toBeDefined()
      })

      it("rejects slug='Bad Slug With Spaces' (uppercase + spaces)", async () => {
        const body = validBody({ slug: 'Bad Slug With Spaces' })
        const result = await service.from('blogs').insert(body)
        expect(result.error).toMatchObject({ message: expect.stringContaining('blogs_slug_format_check') })
      })

      it("accepts canonical_url='/compare/buildium' override", async () => {

        insertedSlugs.push(slug)

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-06-01 | Tampering | `blogs.content` via direct INSERT bypassing pre-flight | mitigate | 9 BEFORE-INSERT validation gates in `validate_blog_post()` trigger; strict `RAISE EXCEPTION` with `ERRCODE='23514'` for any violation |
| T-06-02 | Tampering | `blogs.slug` malformed/exploit (e.g., script injection in URL path) | mitigate | `blogs_slug_format_check` CHECK constraint enforcing `^[a-z][a-z0-9]*(-[a-z0-9]+)*$` length 3–120 (leading-letter requirement closes numeric-only slug class) |
| T-06-03 | Tampering | `blogs.status` set to non-enum value | mitigate | Existing CHECK extended to 4-value enum: `('draft','in-review','published','archived')` |
| T-06-04 | Information Disclosure | Draft / in-review rows visible to anon clients | mitigate | RLS policy `blogs_select_published` already gates anon SELECT to `status='published'` only; integration test pins this |
| T-06-05 | Elevation of Privilege | `validate_blog_post()` callable by anon for side effects | mitigate | `REVOKE EXECUTE ON FUNCTION public.validate_blog_post() FROM anon, authenticated`; SECURITY DEFINER + `SET search_path = public` |
| T-06-06 | Repudiation | Bulk-delete of Phase-1 draft rows without audit trail | accept | One-time cleanup; rows already verified broken; DELETE migration RAISE NOTICEs the count for the migration log |
| T-06-07 | Denial of Service | Validation trigger cost on bulk INSERT | accept | All 9 gates use simple regex/length/position ops; <1ms per row; trigger only fires on transitions to `in-review`/`published` (small volume) |
| T-06-23 | Tampering | Malicious canonical_url value (e.g., XSS via javascript:) injected into post page metadata | mitigate | `canonical_url` is rendered via Next.js `Metadata.alternates.canonical` which serializes via the framework's URL-safe rules; Plan 06-03 preflight should sanity-check that values start with `/` or `https://` before INSERT |
