```typescript
export function calculateAnnualSavings(monthlyPrice: number): number {
  const yearlyPrice = monthlyPrice * 10;   // 2 months free
  const monthlyCost = monthlyPrice * 12;
  return monthlyCost - yearlyPrice;         // === monthlyPrice * 2
}
```
