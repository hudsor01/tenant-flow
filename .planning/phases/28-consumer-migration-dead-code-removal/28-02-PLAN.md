**Step B: Create `src/components/tenants/invitation-table-row.tsx`** (mandatory extraction per 300-line rule)

Create a new file with the InvitationTableRow component. This component receives a single invitation and renders it as a `<tr>` matching TenantTableRow's column structure exactly (8 columns, same `px-4 py-2` cell padding).

```typescript
'use client'

import { useState } from 'react'
import { MoreHorizontal, RefreshCw, Link as LinkIcon, XCircle } from 'lucide-react'
import { formatDistanceToNow } from 'date-fns'
import { toast } from 'sonner'
import { Checkbox } from '#components/ui/checkbox'
import { Button } from '#components/ui/button'
import { Badge } from '#components/ui/badge'
import {
  DropdownMenu, DropdownMenuContent, DropdownMenuItem,
  DropdownMenuSeparator, DropdownMenuTrigger
} from '#components/ui/dropdown-menu'
import { ConfirmDialog } from '#components/ui/confirm-dialog'
import { useResendInvitationMutation, useCancelInvitationMutation } from '#hooks/api/use-tenant-invite-mutations'
import type { TenantSectionInvitation } from '#types/sections/tenants'

interface InvitationTableRowProps {
  invitation: TenantSectionInvitation
}

export function InvitationTableRow({ invitation }: InvitationTableRowProps) {
  const [cancelDialogOpen, setCancelDialogOpen] = useState(false)
  const resendMutation = useResendInvitationMutation()
  const cancelMutation = useCancelInvitationMutation()

  function handleResend(): void {
    resendMutation.mutate(invitation.id)
    toast.success(`Invitation resent to ${invitation.email}`)
  }

  async function handleCopyLink(): Promise<void> {
    if (invitation.invitationUrl) {
      await navigator.clipboard.writeText(invitation.invitationUrl)
      toast.success('Invitation link copied to clipboard')
    }
  }

  function handleCancelConfirm(): void {
    cancelMutation.mutate(invitation.id)
    toast.success(`Invitation to ${invitation.email} cancelled`)
    setCancelDialogOpen(false)
  }

  return (
    <>
      <tr className="hover:bg-muted/50 transition-colors">
        {/* Checkbox - disabled */}
        <td className="px-4 py-2">
          <div className="opacity-50 pointer-events-none">
            <Checkbox disabled aria-disabled="true" aria-label="Cannot select invitation" />
          </div>
        </td>
        {/* Name - shows email (D-03) */}
        <td className="px-4 py-2">
          <span className="text-foreground font-medium">{invitation.email}</span>
        </td>
        {/* Email */}
        <td className="px-4 py-2">
          <span className="text-sm text-muted-foreground">--</span>
        </td>
        {/* Phone */}
        <td className="px-4 py-2">
          <span className="text-sm text-muted-foreground">--</span>
        </td>
        {/* Property */}
        <td className="px-4 py-2">
          <span className="text-sm text-muted-foreground">--</span>
        </td>
        {/* Status + relative timestamp */}
        <td className="px-4 py-2">
          <div className="flex flex-col gap-1">
            <Badge variant={invitation.status === 'pending' ? 'warning' : 'info'}>
              {invitation.status === 'pending' ? 'Pending' : 'Sent'}
            </Badge>
            <span className="text-xs text-muted-foreground">
              Sent {formatDistanceToNow(new Date(invitation.sentAt), { addSuffix: true })}
            </span>
          </div>
        </td>
        {/* Lease */}
        <td className="px-4 py-2">
          <span className="text-sm text-muted-foreground">--</span>
        </td>
        {/* Actions dropdown (D-05) */}
        <td className="px-4 py-2">
          <DropdownMenu>
            <DropdownMenuTrigger asChild>
              <Button variant="ghost" size="icon" className="size-8" aria-label={`Actions for invitation to ${invitation.email}`}>
                <MoreHorizontal className="size-4" />
              </Button>
            </DropdownMenuTrigger>
            <DropdownMenuContent align="end">
              <DropdownMenuItem onClick={handleResend}>
                <RefreshCw className="size-4 mr-2" />
                Resend Invitation
              </DropdownMenuItem>
              {invitation.invitationUrl && (
                <DropdownMenuItem onClick={handleCopyLink}>
                  <LinkIcon className="size-4 mr-2" />
                  Copy Invitation Link
                </DropdownMenuItem>
              )}
              <DropdownMenuSeparator />
              <DropdownMenuItem
                className="text-destructive focus:text-destructive"
                onClick={() => setCancelDialogOpen(true)}
              >
                <XCircle className="size-4 mr-2" />
                Cancel Invitation
              </DropdownMenuItem>
            </DropdownMenuContent>
          </DropdownMenu>
        </td>
      </tr>

      {/* Cancel confirmation dialog (D-06) */}
      <ConfirmDialog
        open={cancelDialogOpen}
        onOpenChange={setCancelDialogOpen}
        title="Cancel Invitation"
        description={`Cancel invitation to ${invitation.email}? They will no longer be able to use this link to accept.`}
        confirmText="Cancel Invitation"
        confirmVariant="destructive"
        loading={cancelMutation.isPending}
        onConfirm={handleCancelConfirm}
      />
    </>
  )
}
```

The component calls `useResendInvitationMutation` and `useCancelInvitationMutation` directly -- no prop-passing of mutation callbacks from page.tsx. This keeps the component self-contained and avoids prop drilling.
