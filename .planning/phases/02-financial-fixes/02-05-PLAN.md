```sql
-- Backfill onboarding_completed_at for owners who completed Stripe onboarding
-- but the timestamp was lost due to the account.updated overwrite bug
update public.users u
set onboarding_completed_at = now()
where u.onboarding_completed_at is null
  and u.stripe_account_id is not null
  and exists (
    select 1 from stripe.accounts sa
    where sa.id = u.stripe_account_id
    and (sa.attrs->>'charges_enabled')::boolean = true
  );
```
