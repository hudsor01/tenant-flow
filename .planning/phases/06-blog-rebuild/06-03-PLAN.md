The Edge Function MUST accept this payload from n8n:
```typescript
interface IngestPayload {
  title: string                // 5-100 chars
  slug: string                 // ^[a-z][a-z0-9]*(-[a-z0-9]+)*$, length 3..120
  excerpt: string              // 80-200 chars
  content: string              // 1,200-3,000 words, markdown

  meta_description: string     // 50-160 chars
  og_image_url?: string        // optional override; nominal default is /api/og/blog/{slug}
  canonical_url?: string       // optional <link rel="canonical"> override; must start with '/' or 'https://'; threaded into blogs.canonical_url INSERT (Blocker-#1)
}
```

<!-- HMAC signature spec -->
n8n includes header `x-n8n-signature: sha256(N8N_WEBHOOK_SECRET + body_json)`.
- The body_json is the EXACT bytes of the request body (UTF-8 JSON, no normalization).
- Compute server-side using Web Crypto API (HMAC-SHA256) and compare in constant time.
- Mismatch â†’ 401 with `{error: 'unauthorized'}` (generic).
- Locks: secret stored in Supabase Vault via `secrets create N8N_WEBHOOK_SECRET <value>`. Edge Function reads via `Deno.env.get('N8N_WEBHOOK_SECRET')`.

<!-- Response shapes -->
- 201 Created: `{id: string, slug: string, status: 'in-review', blog_url: string}` â€” `blog_url` is `${FRONTEND_URL}/blog/${slug}` (post will be public after editorial flip).
- 400 Bad Request: `{error: 'validation_failed', gate_failures: [{gate: string, message: string}]}` â€” at least one gate failed.
- 401 Unauthorized: `{error: 'unauthorized'}` â€” signature mismatch or missing.
- 409 Conflict: `{error: 'slug_collision', slug: string}` â€” slug already exists.
- 500 Internal Server Error: generic per CLAUDE.md (`{error: 'An error occurred'}`); full details in Sentry.

    ```typescript
    // runs 9 validation gates as defense-in-depth (DB trigger is primary), inserts as
    // status='in-review'. Per Phase 6 / BLOG-03.

    import { createClient } from 'https://esm.sh/@supabase/supabase-js@2'
    import { handleCorsOptions, getCorsHeaders } from '../_shared/cors.ts'
    import { captureWebhookError, errorResponse, logEvent } from '../_shared/errors.ts'
    import { validateEnv } from '../_shared/env.ts'

    interface IngestPayload {
      title: string
      slug: string
      excerpt: string
      content: string
      category: 'lease-law' | 'tax-prep' | 'tenant-screening' | 'maintenance' | 'software-vault'
      meta_description: string
      og_image_url?: string
      canonical_url?: string
    }

    interface GateFailure {
      gate: string
      message: string
    }

    const VALID_CATEGORIES = ['lease-law', 'tax-prep', 'tenant-screening', 'maintenance', 'software-vault'] as const
    // Slug regex aligned with Plan 06-01 Blocker-#4 fix: must start with a letter.
    const SLUG_REGEX = /^[a-z][a-z0-9]*(-[a-z0-9]+)*$/

    const BANLIST = [
      'rent collection', 'online rent', 'autopay', 'auto-pay', 'tenant portal',
      'automated rent', 'collect rent', 'rent processing', 'pay rent online',
      'online payments', 'online rent payment', 'rent collection software',
      'tenants can pay', 'pay rent through',
      'automated workflow', 'rent tracking', 'mobile app access',
      'record rent', 'paid rent', 'pay rent',
    ] as const

    function jsonResponse(req: Request, status: number, body: unknown): Response {
      return new Response(JSON.stringify(body), {
        status,
        headers: {
          'Content-Type': 'application/json',
          ...getCorsHeaders(req),
        },
      })
    }

    async function verifyHmac(secret: string, body: string, signature: string | null): Promise<boolean> {
      if (!signature) return false
      const enc = new TextEncoder()
      const key = await crypto.subtle.importKey(
        'raw',
        enc.encode(secret),
        { name: 'HMAC', hash: 'SHA-256' },
        false,
        ['sign'],
      )
      const sigBuf = await crypto.subtle.sign('HMAC', key, enc.encode(body))
      const expected = Array.from(new Uint8Array(sigBuf))
        .map((b) => b.toString(16).padStart(2, '0'))
        .join('')
      // Constant-time comparison
      if (expected.length !== signature.length) return false
      let diff = 0
      for (let i = 0; i < expected.length; i++) {
        diff |= expected.charCodeAt(i) ^ signature.charCodeAt(i)
      }
      return diff === 0
    }

    function countWords(s: string): number {
      return s.trim().split(/\s+/).filter(Boolean).length
    }

    function countH2(s: string): number {
      return (s.match(/^## /gm) ?? []).length
    }

    function countMatches(s: string, pattern: RegExp): number {
      return (s.match(pattern) ?? []).length
    }

    function isValidCanonicalUrl(s: string): boolean {
      return s.startsWith('/') || s.startsWith('https://')
    }

    function runGates(p: IngestPayload): GateFailure[] {
      const failures: GateFailure[] = []
      const wc = countWords(p.content)
      if (wc < 1200 || wc > 3000) failures.push({ gate: 'word_count', message: `out of range: ${wc} (must be 1200..3000)` })

      const h2 = countH2(p.content)
      if (h2 < 4 || h2 > 10) failures.push({ gate: 'h2_count', message: `out of range: ${h2} (must be 4..10)` })

      if (!/landlord/i.test(p.content)) failures.push({ gate: 'persona_phrase', message: 'content must mention "landlord"' })

      if (!SLUG_REGEX.test(p.slug) || p.slug.length < 3 || p.slug.length > 120) {
        failures.push({ gate: 'slug_pattern', message: `slug invalid: ${p.slug} (must start with a letter, kebab-case, length 3..120)` })
      }

      if (!p.meta_description || p.meta_description.length < 50 || p.meta_description.length > 160) {
        failures.push({ gate: 'meta_length', message: `meta_description out of range: ${p.meta_description?.length ?? 0} (must be 50..160)` })
      }

      if (!p.excerpt || p.excerpt.length < 80 || p.excerpt.length > 200) {
        failures.push({ gate: 'excerpt_length', message: `excerpt out of range: ${p.excerpt?.length ?? 0} (must be 80..200)` })
      }

      if (!VALID_CATEGORIES.includes(p.category as typeof VALID_CATEGORIES[number])) {
        failures.push({ gate: 'category', message: `category not in enum: ${p.category}` })
      }

      const lowerContent = p.content.toLowerCase()
      for (const phrase of BANLIST) {
        if (lowerContent.includes(phrase)) {
          failures.push({ gate: 'banlist', message: `phrase found: ${phrase}` })
          break // one is enough; n8n flow can fix and resubmit
        }
      }

      const docusealCount = countMatches(p.content, /DocuSeal/gi)
      if (docusealCount > 1) {
        failures.push({ gate: 'docuseal_mention', message: `count too high: ${docusealCount} (max 1 per Phase 4 COPY-04)` })
      }

      // canonical_url shape gate (when provided)
      if (p.canonical_url !== undefined) {
        if (typeof p.canonical_url !== 'string' || !isValidCanonicalUrl(p.canonical_url)) {
          failures.push({ gate: 'canonical_url_format', message: `canonical_url must start with "/" or "https://"; got: ${p.canonical_url}` })
        }
      }

      return failures
    }

    Deno.serve(async (req) => {
      const corsResponse = handleCorsOptions(req)
      if (corsResponse) return corsResponse

      if (req.method !== 'POST') {
        return jsonResponse(req, 405, { error: 'method_not_allowed' })
      }

      try {
        const env = validateEnv({
          required: ['SUPABASE_URL', 'SUPABASE_SERVICE_ROLE_KEY', 'N8N_WEBHOOK_SECRET', 'FRONTEND_URL'],
        })

        const bodyText = await req.text()
        const signature = req.headers.get('x-n8n-signature')

        const valid = await verifyHmac(env.N8N_WEBHOOK_SECRET, bodyText, signature)
        if (!valid) {
          logEvent('n8n-blog-ingest: signature mismatch', { hasSignature: !!signature })
          return jsonResponse(req, 401, { error: 'unauthorized' })
        }

        let payload: IngestPayload
        try {
          payload = JSON.parse(bodyText)
        } catch {
          return jsonResponse(req, 400, { error: 'malformed_json' })
        }

        // Required-field check before gate run
        const required: (keyof IngestPayload)[] = ['title', 'slug', 'excerpt', 'content', 'category', 'meta_description']
        const missing = required.filter((k) => !payload[k] || typeof payload[k] !== 'string')
        if (missing.length > 0) {
          return jsonResponse(req, 400, {
            error: 'validation_failed',
            gate_failures: missing.map((k) => ({ gate: 'required_field', message: `missing or non-string: ${k}` })),
          })
        }

        const failures = runGates(payload)
        if (failures.length > 0) {
          return jsonResponse(req, 400, { error: 'validation_failed', gate_failures: failures })
        }

        const supabase = createClient(env.SUPABASE_URL, env.SUPABASE_SERVICE_ROLE_KEY)

        const insertRow: Record<string, unknown> = {
          title: payload.title,
          slug: payload.slug,
          excerpt: payload.excerpt,
          content: payload.content,
          category: payload.category,
          meta_description: payload.meta_description,
          featured_image: payload.og_image_url ?? null,
          status: 'in-review',
        }
        // Thread canonical_url into INSERT only when provided. Plan 06-02's
        // generateMetadata() emits <link rel="canonical"> when this column is non-null.
        if (payload.canonical_url !== undefined) {
          insertRow.canonical_url = payload.canonical_url
        }

        const { data, error } = await supabase
          .from('blogs')
          .insert(insertRow)
          .select('id, slug, status, canonical_url')
          .single()

        if (error) {
          // Slug collision (unique violation) â†’ 409
          if (error.code === '23505') {
            return jsonResponse(req, 409, { error: 'slug_collision', slug: payload.slug })
          }
          // CHECK violation that survived our preflight (e.g., DB trigger drift) â†’ 400 with gate name
          if (error.code === '23514') {
            return jsonResponse(req, 400, {
              error: 'validation_failed',
              gate_failures: [{ gate: 'db_trigger', message: error.message }],
            })
          }
          throw error
        }

        logEvent('n8n-blog-ingest: insert success', { id: data?.id, slug: data?.slug, has_canonical: data?.canonical_url !== null })

        return jsonResponse(req, 201, {
          id: data?.id,
          slug: data?.slug,
          status: data?.status,
          canonical_url: data?.canonical_url,
          blog_url: `${env.FRONTEND_URL}/blog/${data?.slug}`,
        })
      } catch (err) {
        captureWebhookError(err, { fn: 'n8n-blog-ingest' })
        return errorResponse(req, 500, err, { fn: 'n8n-blog-ingest' })
      }
    })
    ```
