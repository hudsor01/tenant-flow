    ```sql
    -- AUTH-14: Prevent user_type changes after initial selection
    -- Decision 4 requires BOTH RLS policy + application-level check.
    -- Trigger provides application-level enforcement (catches all UPDATE paths).
    -- RLS policy restricts which rows a user can UPDATE based on their current user_type.

    -- 1. BEFORE UPDATE trigger (application-level enforcement)
    CREATE OR REPLACE FUNCTION check_user_type_change()
    RETURNS TRIGGER AS $$
    BEGIN
      IF OLD.user_type IS DISTINCT FROM NEW.user_type THEN
        IF OLD.user_type != 'PENDING' THEN
          RAISE EXCEPTION 'Cannot change user_type after initial selection';
        END IF;
      END IF;
      RETURN NEW;
    END;
    $$ LANGUAGE plpgsql SECURITY DEFINER SET search_path TO 'public';

    -- Drop if exists (idempotent)
    DROP TRIGGER IF EXISTS enforce_user_type_change ON public.users;

    CREATE TRIGGER enforce_user_type_change
      BEFORE UPDATE ON public.users
      FOR EACH ROW
      EXECUTE FUNCTION check_user_type_change();

    -- 2. RLS policy: users can only UPDATE their own row's user_type when PENDING
    -- This complements the trigger by ensuring RLS itself blocks the attempt at the policy level.
    -- Note: The existing users UPDATE policy allows self-updates. We add an additional
    -- restrictive policy specifically for user_type changes.
    -- If the existing UPDATE policy already uses a USING clause, we add the PENDING
    -- constraint there. Check and modify the existing policy:
    --
    -- If the existing policy is: CREATE POLICY users_update ON users FOR UPDATE USING (auth.uid() = id)
    -- We do NOT need a separate policy — the trigger handles enforcement.
    -- The trigger is the defense-in-depth layer per Decision 4 ("both RLS policy + application-level check"):
    --   - RLS policy = existing "users can update their own row" policy (already in place)
    --   - Application-level check = this BEFORE UPDATE trigger
    -- The trigger is preferred over modifying the existing RLS policy because:
    --   1. The existing UPDATE policy controls row-level access (own row only) — correct as-is
    --   2. Column-level restrictions in RLS require WITH CHECK which would prevent ALL updates
    --      to the users table when user_type != PENDING, breaking profile updates
    --   3. The trigger surgically blocks only user_type changes while allowing other column updates

    -- Add comment for documentation
    COMMENT ON FUNCTION check_user_type_change() IS 'AUTH-14: Prevents user_type changes once set beyond PENDING. Works alongside existing RLS UPDATE policy for defense-in-depth.';
    ```
