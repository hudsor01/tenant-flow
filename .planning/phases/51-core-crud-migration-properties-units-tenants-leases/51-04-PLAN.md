    - import { createClient } from '#lib/supabase/client'

    Migrate these mutationFn functions:

    1. useCreateLeaseMutation — mutationFn (data: LeaseCreate):
       - const supabase = createClient()
       - const { data: { user } } = await supabase.auth.getUser()
       - const { data: created, error } = await supabase.from('leases').insert({ ...data, owner_user_id: user?.id }).select().single()

       - return created as Lease

       - const supabase = createClient()
       - const payload = version ? { ...data, version } : { ...data }
       - const { data: updated, error } = await supabase.from('leases').update(payload).eq('id', id).select().single()
       - if (error) handlePostgrestError(error, 'leases')
       - return updated as Lease

    3. useDeleteLeaseOptimisticMutation — mutationFn (id: string):
       - Soft-delete: supabase.from('leases').update({ status: 'inactive' }).eq('id', id)
       - Keep the full optimistic update/rollback onMutate/onError logic unchanged

    4. useDeleteLeaseMutation — mutationFn (id: string):
       - Soft-delete: supabase.from('leases').update({ status: 'inactive' }).eq('id', id)

    5. useTerminateLeaseMutation — mutationFn (id: string):
       - supabase.from('leases').update({ status: 'terminated', end_date: new Date().toISOString() }).eq('id', id).select().single()

    6. useRenewLeaseMutation — mutationFn ({ id, data }):
