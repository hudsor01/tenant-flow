import type { Metadata } from 'next'
import { createServerClient } from '#lib/supabase/server'
import { DeliverabilityTable } from '#components/admin/deliverability-table'
import { FunnelChartClient } from '#components/admin/funnel-chart'
import { mapDeliverabilityRow } from '#hooks/api/query-keys/deliverability-keys'
import { mapFunnelStats } from '#hooks/api/query-keys/funnel-keys'
import type { DeliverabilityStats, FunnelStats } from '#types/analytics'

export const metadata: Metadata = { title: 'Admin Analytics | TenantFlow' }

export default async function AdminAnalyticsPage() {
  const supabase = await createServerClient()
  const [deliverabilityResult, funnelResult] = await Promise.all([
    supabase.rpc('get_deliverability_stats', { p_days: 30 }),
    supabase.rpc('get_funnel_stats', {}),
  ])

  const deliverability: DeliverabilityStats[] = Array.isArray(deliverabilityResult.data)
    ? deliverabilityResult.data.map(row => mapDeliverabilityRow(row as Record<string, unknown>))
    : []
  const funnel: FunnelStats | null = funnelResult.data
    ? mapFunnelStats(funnelResult.data as Record<string, unknown>)
    : null

  return (
    <div className="max-w-7xl mx-auto px-6 lg:px-8 py-8 space-y-8">
      <div>
        <h2 className="text-2xl font-semibold text-foreground">Platform Analytics</h2>
        <p className="text-sm text-muted-foreground">Email deliverability (rolling 30 days) and owner onboarding funnel.</p>
      </div>
      <section aria-labelledby="deliverability-heading">
        <h3 id="deliverability-heading" className="text-lg font-semibold mb-4 text-foreground">Email Deliverability</h3>
        <DeliverabilityTable data={deliverability} />
      </section>
      <section aria-labelledby="funnel-heading">
        <h3 id="funnel-heading" className="text-lg font-semibold mb-4 text-foreground">Onboarding Funnel</h3>

      </section>
    </div>
  )
}

/**
 * Onboarding Funnel Analytics Query Options
 *
 * Reads `get_funnel_stats(p_from date, p_to date)` RPC.
 * Cohort semantics: owners who signed up in [p_from, p_to] — their funnel is
 * tracked to the present day. See 44-DECISIONS.md D2.
 */
import { queryOptions } from '@tanstack/react-query'
import { createClient } from '#lib/supabase/client'
import { handlePostgrestError } from '#lib/postgrest-error-handler'
import type { FunnelStats, FunnelStep, FunnelStepName } from '#types/analytics'

const ALLOWED_STEPS: readonly FunnelStepName[] = [
  'signup',
  'first_property',
  'first_tenant_invited',
  'first_rent_collected',
] as const

function isFunnelStepName(value: unknown): value is FunnelStepName {
  return typeof value === 'string' && (ALLOWED_STEPS as readonly string[]).includes(value)
}

export function mapFunnelStats(raw: Record<string, unknown>): FunnelStats {
  const stepsRaw = Array.isArray(raw.steps) ? raw.steps : []
  const steps: FunnelStep[] = stepsRaw
    .map((s): FunnelStep | null => {
      if (!s || typeof s !== 'object') return null
      const row = s as Record<string, unknown>
      if (!isFunnelStepName(row.step_name)) return null
      const conv = row.conversion_rate_from_prev
      return {
        stepName: row.step_name,
        count: Number(row.count ?? 0),
        conversionRateFromPrev: conv === null || conv === undefined ? null : Number(conv),
      }
    })
    .filter((s): s is FunnelStep => s !== null)
  return {
    cohortFrom: typeof raw.cohort_from === 'string' ? raw.cohort_from : '',
    cohortTo: typeof raw.cohort_to === 'string' ? raw.cohort_to : '',
    steps,
  }
}

export const funnelKeys = {
  all: ['funnel'] as const,
  stats: (from: string, to: string) => [...funnelKeys.all, 'stats', from, to] as const,
}

export const funnelQueries = {
  stats: (from: string, to: string) =>
    queryOptions({
      queryKey: funnelKeys.stats(from, to),
      queryFn: async (): Promise<FunnelStats | null> => {
        const supabase = createClient()
        const { data, error } = await supabase.rpc('get_funnel_stats', { p_from: from, p_to: to })
        if (error) handlePostgrestError(error, 'funnel-stats')
        if (!data || typeof data !== 'object') return null
        return mapFunnelStats(data as Record<string, unknown>)
      },
      staleTime: 5 * 60 * 1000,
      gcTime: 10 * 60 * 1000,
    }),
}

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Browser → proxy.ts | Unauthenticated HTTP requests must not reach `/admin/*` page handlers. Proxy runs at the edge, enforces route-level auth. |
| proxy.ts → (admin)/layout.tsx | Even if proxy is bypassed (internal Next.js routing, edge cases), the server-component layout MUST re-validate. Defense in depth. |
| layout.tsx → PostgREST/RPC | Even if layout is bypassed (direct RPC call from browser via supabase-js), the DB-level `is_admin()` guard MUST filter data. This is the only layer that survives all bypasses. |
| Recipient emails → deliverability table | Resend webhook payloads carry arbitrary recipient email strings. Rendering unsanitized user-controlled text in the table is a potential XSS vector. |
| Session cookie → redirect response | When proxy redirects a non-admin user, the response MUST carry forward Supabase session cookies. Dropped cookies cause a follow-up `/dashboard` request to bounce back to `/login`. |

| T-44-03-03 | Elevation of Privilege | (admin)/layout.tsx gate | mitigate | Layout server component re-checks `user.app_metadata.user_type`; redirect to `/dashboard` on non-admin. Second-chance guard if proxy misses (e.g., internal Next.js routing, SSR revalidation paths). |
| T-44-03-04 | Tampering | client-side admin check without server verification | mitigate | No client-side `isAdmin` state exists in this plan. Admin verification is server-side only: proxy (edge), layout (server component), RPC (DB). Client components receive data pre-fetched on the server. |
| T-44-03-05 | Information Disclosure | XSS via recipient_email in deliverability table | mitigate | Deliverability table renders `template_tag` (server-controlled category string — never user input). | T-44-03-06 | Denial of Service | redirect loop if admin DB check fails | mitigate | Layout check uses `user.app_metadata.user_type` (JWT claim, no DB round-trip per D5 / Anti-Pattern in 44-RESEARCH.md line 797).
