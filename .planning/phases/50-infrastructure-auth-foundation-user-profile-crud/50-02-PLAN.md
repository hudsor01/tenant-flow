Feature flag pattern for each queryFn:
```typescript
import { isPostgrestEnabled } from '#lib/postgrest-flag'
import { createClient } from '#lib/supabase/client'

queryFn: async () => {
  if (isPostgrestEnabled()) {
    const supabase = createClient()
    const { data, error } = await supabase
      .from('users')
      .select('id, email, first_name, last_name, full_name, phone, avatar_url, role, updated_at')
      .single()
    if (error) throw error
    return data
  }
  return apiRequest<UserProfile>('/api/v1/users/profile')
}
```
