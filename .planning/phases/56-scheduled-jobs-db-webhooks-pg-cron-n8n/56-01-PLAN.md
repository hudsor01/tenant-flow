**Section 1 — Enable pg_cron extension:**
```sql
create extension if not exists pg_cron with schema extensions;
```
Note: pg_cron must be enabled in the `extensions` schema on Supabase (not `pg_catalog`). Add a comment explaining this is required for scheduled jobs.

**Section 2 — Expand leases.lease_status CHECK constraint to allow 'expired':**
The current constraint only allows: `draft, pending_signature, active, ended, terminated`.
We need to add `expired` for the nightly expiry job.
```sql
alter table public.leases
  drop constraint leases_lease_status_check;

alter table public.leases
  add constraint leases_lease_status_check
  check (lease_status in (
    'draft', 'pending_signature', 'active', 'ended', 'terminated', 'expired'
  ));

comment on constraint leases_lease_status_check on public.leases is
  'Valid values: draft, pending_signature, active, ended, terminated, expired. expired set by pg_cron nightly job.';
```

**Section 3 — Expand rent_payments.status CHECK constraint to allow 'late' and 'severely_delinquent':**
Current constraint allows: `pending, processing, succeeded, failed, canceled`.
Add `late` (grace period exceeded, fees accruing) and `severely_delinquent` (14+ days past due).
```sql
alter table public.rent_payments
  drop constraint rent_payments_status_check;

alter table public.rent_payments
  add constraint rent_payments_status_check
  check (status in (
    'pending', 'processing', 'succeeded', 'failed', 'canceled', 'late', 'severely_delinquent'
  ));

comment on constraint rent_payments_status_check on public.rent_payments is
  'Valid values: pending, processing, succeeded, failed, canceled, late (>3 days past due), severely_delinquent (>14 days past due).';
```

**Section 4 — Create `late_fees` table:**
This tracks individual daily fee records assessed against overdue rent payments.
```sql
create table if not exists public.late_fees (
  id             uuid    primary key default extensions.uuid_generate_v4(),
  rent_payment_id uuid   not null references public.rent_payments(id) on delete cascade,
  lease_id       uuid    not null references public.leases(id) on delete cascade,
  fee_amount     integer not null,               -- in cents
  days_overdue   integer not null,               -- how many days past due when assessed
  assessed_date  date    not null default current_date,
  created_at     timestamptz not null default now(),
  constraint late_fees_fee_amount_check check (fee_amount > 0),
  constraint late_fees_days_overdue_check check (days_overdue > 0)
);

comment on table public.late_fees is 'Daily late fee records assessed by pg_cron against overdue rent payments.';
comment on column public.late_fees.fee_amount is 'Fee amount in cents (e.g. 5000 = $50.00).';
comment on column public.late_fees.days_overdue is 'Number of days past due_date when this fee was assessed (>3 grace period).';

-- Prevent duplicate fee for same payment on same day
create unique index if not exists late_fees_unique_per_day
  on public.late_fees (rent_payment_id, assessed_date);

-- Index for querying fees by payment
create index if not exists idx_late_fees_rent_payment_id
  on public.late_fees using btree (rent_payment_id);

-- Enable RLS
alter table public.late_fees enable row level security;
```

**Section 5 — RLS policies for `late_fees`:**
Late fees are only readable by the property owner (via lease → property_owner_id join) and the tenant (via lease → tenants join). Fees are written by pg_cron (SECURITY DEFINER function bypasses RLS), so INSERT/UPDATE/DELETE policies are restricted.

```sql
-- Owners can view late fees for their properties (join through leases)
create policy "Property owners can view late fees for their leases"
  on public.late_fees
  for select
  to authenticated
  using (
    lease_id in (
      select id from public.leases
      where property_owner_id = (select auth.uid())
    )
  );

-- Tenants can view their own late fees (join through leases → lease_tenants)
create policy "Tenants can view their own late fees"
  on public.late_fees
  for select
  to authenticated
  using (
    lease_id in (
      select lt.lease_id from public.lease_tenants lt
      inner join public.tenants t on t.id = lt.tenant_id
      where t.user_id = (select auth.uid())
    )
  );
```

**Section 6 — Create `lease_reminders` table:**
Dedup queue for lease expiry notifications. pg_cron inserts rows; DB Webhook fires on INSERT; UNIQUE constraint prevents duplicates.
```sql
create table if not exists public.lease_reminders (
  id             uuid    primary key default extensions.uuid_generate_v4(),
  lease_id       uuid    not null references public.leases(id) on delete cascade,
  reminder_type  text    not null,   -- '30_days', '7_days', '1_day'
  sent_at        timestamptz not null default now(),
  created_at     timestamptz not null default now(),
  constraint lease_reminders_reminder_type_check
    check (reminder_type in ('30_days', '7_days', '1_day')),
  -- Idempotency guard: one reminder per (lease, type) ever
  constraint lease_reminders_unique_per_lease_type
    unique (lease_id, reminder_type)
);

comment on table public.lease_reminders is 'Dedup queue for lease expiry reminders. pg_cron inserts with ON CONFLICT DO NOTHING; DB Webhook fires on INSERT to trigger n8n notification workflow.';
comment on column public.lease_reminders.reminder_type is 'Notification threshold: 30_days, 7_days, or 1_day before lease end_date.';

-- Index for checking if reminder already sent
create index if not exists idx_lease_reminders_lease_id
  on public.lease_reminders using btree (lease_id);

-- Enable RLS
alter table public.lease_reminders enable row level security;
```

**Section 7 — RLS policies for `lease_reminders`:**
Lease reminders are internal queue data. Owners can view reminders for their leases (useful for debugging). No INSERT/UPDATE/DELETE from authenticated users — managed by pg_cron SECURITY DEFINER function only.

```sql
create policy "Property owners can view lease reminder history"

  for select
  to authenticated
  using (

      select id from public.leases
      where property_owner_id = (select auth.uid())
    )
  );
```
