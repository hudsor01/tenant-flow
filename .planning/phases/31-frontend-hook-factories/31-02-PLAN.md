```typescript
interface OptimisticConfig<TVariables, TContext> {
  cancel: ReadonlyArray<readonly unknown[]> | ((variables: TVariables) => ReadonlyArray<readonly unknown[]>)
  snapshot: (queryClient: QueryClient, variables: TVariables) => TContext
  rollback: (queryClient: QueryClient, context: TContext, variables: TVariables) => void
  apply?: (queryClient: QueryClient, variables: TVariables) => void
}

interface MutationCallbackConfig<TData, TVariables = unknown, TContext = unknown> {
  invalidate: ReadonlyArray<readonly unknown[]>
  successMessage?: string
  errorContext: string
  updateDetail?: (data: TData) => { queryKey: readonly unknown[]; data: TData }
  removeDetail?: (data: TData, variables: TVariables) => readonly unknown[]
  onSuccessExtra?: (data: TData) => void
  useSuccessHandler?: boolean
  optimistic?: OptimisticConfig<TVariables, TContext>
}

// Tier 1/2 (no optimistic): returns { onSuccess, onError }
// Tier 3 (with optimistic): returns { onMutate, onSuccess, onError, onSettled }
export function createMutationCallbacks<TData, TVariables = unknown, TContext = unknown>(
  queryClient: QueryClient,
  config: MutationCallbackConfig<TData, TVariables, TContext>
): { onSuccess: ...; onError: ...; onMutate?: ...; onSettled?: ... }
```

    ```typescript
              if (context.previousDetail) {

              }
    ```
