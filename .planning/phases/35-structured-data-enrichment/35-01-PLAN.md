2. Convert the default export from a synchronous function to an async function that fetches blog post data for JSON-LD. Add a Supabase query inside the component body (separate from generateMetadata) to get the fields needed for Article schema:
```typescript
export default async function Page({ params }: Props) {
  const { slug } = await params
  const supabase = await createClient()

  // Fetch post data for JSON-LD (Next.js deduplicates with generateMetadata query via ISR cache)
  const { data: post } = await supabase
    .from('blogs')
    .select('title, slug, published_at, updated_at, featured_image, content, reading_time, category, meta_description, excerpt, tags')
    .eq('slug', slug)
    .eq('status', 'published')
    .single()

  // Compute word count from content (per D-02, SCHEMA-08)
  const wordCount = post?.content
    ? post.content.trim().split(/\s+/).length
    : undefined

  // 4-level breadcrumb: Home > Blog > Category > Post Title (per D-05)
  // Uses /blog/category/[cat-slug]/[post-slug] synthetic path so intermediate
  // category URL resolves to the real /blog/category/[category] route
  const categorySlug = post?.category
    ? post.category.toLowerCase().replace(/\s+/g, '-')
    : ''

  const breadcrumbSchema = post
    ? createBreadcrumbJsonLd(
        `/blog/category/${categorySlug}/${slug}`,
        {
          [categorySlug]: post.category ?? categorySlug,
          [slug]: post.title ?? slug
        }
      )
    : null

  // Article schema with Person author (per D-01, D-02, D-03)
  const articleSchema = post
    ? createArticleJsonLd({
        title: post.title,
        slug: post.slug,
        datePublished: post.published_at ?? new Date().toISOString(),
        dateModified: post.updated_at ?? post.published_at ?? undefined,
        authorName: 'Richard Hudson',
        image: post.featured_image ?? undefined,
        wordCount,
        keywords: Array.isArray(post.tags) ? (post.tags as string[]) : undefined,
        description: post.meta_description ?? post.excerpt ?? undefined
      })
    : null

  return (
    <>
      {breadcrumbSchema && <JsonLdScript schema={breadcrumbSchema} />}
      {articleSchema && <JsonLdScript schema={articleSchema} />}
      <BlogPostPage />
    </>
  )
}
```
